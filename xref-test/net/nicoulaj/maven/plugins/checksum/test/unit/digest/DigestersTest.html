38  /**
39   * Tests for each implementation of {@link net.nicoulaj.maven.plugins.checksum.digest.FileDigester}.

45  @RunWith( Parameterized.class )
46  public class DigestersTest
47  {
48      /**
49       * The {@link net.nicoulaj.maven.plugins.checksum.digest.FileDigester} tested.
50       */
51      private FileDigester digester;
52  
53      /**
54       * Rule used to specify per-test expected exceptions.
55       */
56      @Rule
57      public ExpectedException exception = ExpectedException.none();
58  
59      /**
60       * Generate the list of arguments with which the test should be run.
61       *
62       * @return the list of tested {@link net.nicoulaj.maven.plugins.checksum.digest.FileDigester} implementations.
63       */
64      @Parameterized.Parameters
65      public static Collection<Object[]> getTestParameters()
66      {
67          List<Object[]> data = new ArrayList<Object[]>();
68          for ( String algorithm : net.nicoulaj.maven.plugins.checksum.Constants.SUPPORTED_ALGORITHMS )
69          {
70              data.add( new String[]{algorithm} );
71          }
72          return data;
73      }
74  
75      /**
76       * Build a new {@link DigestersTest}.
77       *
78       * @param algorithm the target checksum algorithm to run the test for.
79       * @throws NoSuchAlgorithmException should never happen.
80       */
81      public DigestersTest( String algorithm ) throws NoSuchAlgorithmException
82      {
83          this.digester = DigesterFactory.getInstance().getFileDigester( algorithm );
84      }
85  
86      /**
87       * Assert the algorithm name is not null/empty.
88       */
89      @Test
90      public void testAlgorithmNameDefined()
91      {
92          String algorithmName = digester.getAlgorithm();
93          Assert.assertNotNull( "The algorithm name is null.", algorithmName );
94          Assert.assertTrue( "The algorithm name is empty.", algorithmName.length() > 0 );
95      }
96  
97      /**
98       * Assert the file name extension is not null/empty.
99       */
100     @Test
101     public void testFilenameExtensionDefined()
102     {
103         String filenameExtension = digester.getFileExtension();
104         Assert.assertNotNull( "The file name extension is null.", filenameExtension );
105         Assert.assertTrue( "The file name extension is empty.", filenameExtension.length() > 0 );
106     }
107 
108     /**
109      * Check the calculated checksum for a specific file is valid against a pre-calculated checksum.
110      *
111      * @throws DigesterException if there was a problem while calculating the checksum.
112      * @throws IOException       if there was a problem reading the file containing the pre-calculated checksum.
113      * @see net.nicoulaj.maven.plugins.checksum.digest.FileDigester#calculate(java.io.File)
114      */
115     @Test
116     public void testCalculate() throws DigesterException, IOException
117     {
118         List<File> testFiles = FileUtils.getFiles( new File( Constants.SAMPLE_FILES_PATH ), null, null );
119         for ( File testFile : testFiles )
120         {
121             String calculatedHash = digester.calculate( testFile );
122             String correctHash = FileUtils.fileRead( Constants.SAMPLE_FILES_HASHCODES_PATH + File.separator
123                                                      + testFile.getName() + digester.getFileExtension() );
124             Assert.assertEquals( "The calculated " + digester.getAlgorithm() + " hashcode for "
125                                  + testFile.getName() + " is incorrect.", correctHash, calculatedHash );
126         }
127     }
128

131      *
132      * @throws DigesterException should always happen.

134      */
135     @Test
136     public void testCalculateExceptionThrownOnFileNotFound() throws DigesterException
137     {
138         exception.expect( DigesterException.class );
139         digester.calculate( new File( "some/path/that/does/not/exist" ) );
140     }
