16  package net.nicoulaj.maven.plugins.checksum.execution.target;
17  
18  import java.io.File;
19  
20  /**
21   * An {@link ExecutionTarget} is an output target for calculated digests.
22   *
23   * @author <a href="mailto:julien.nicoulaud@gmail.com">Julien Nicoulaud</a>
24   * @since 1.0
25   */
26  public interface ExecutionTarget
27  {
28      /**
29       * Initialize the target.
30       *
31       * <p>Should be called before first call to {@link #write(String, java.io.File, String)}.</p>
32       *
33       * @throws ExecutionTargetInitializationException
34       *          if an error occured while initializing the target.
35       */
36      void init() throws ExecutionTargetInitializationException;
37  
38      /**
39       * Write the hashcode calculated for the given file and algorithm to the target.
40       *
41       * @param digest    the digest to write.
42       * @param file      the file for which the digest was calculated.
43       * @param algorithm the algorithm used to calculate the digest.
44       * @throws ExecutionTargetWriteException if an error occured while writing to the target.
45       */
46      void write( String digest, File file, String algorithm ) throws ExecutionTargetWriteException;
47  
48      /**
49       * Close the target.
50       *
51       * <p>Should be called after last call to {@link #write(String, java.io.File, String)}.</p>
52       *
53       * @throws ExecutionTargetCloseException if an error occured while closing the target.
54       */
55      void close() throws ExecutionTargetCloseException;
56  }
